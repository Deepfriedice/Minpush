	' '

state
	{		start state / label
	:		start / end state body
	$		state restart point
	?		start conditional switch state
	;		end switch state
	}		end state

literals
	`		character
	b		start binary
	B		end binary
	d		start decimal
	D		end decimal
	h		start hexadecimal
	H		end hexadecimal

stack manipulation
	t		trim (drop)
	c		copy (dup)
	s		swap
	r		rotate
	R		reverse rotate

arithmetic operations
	~		arithmetic negate
	+		addition
	-		subtraction
	*		multiplication
	/		division
	%		modulo

comparison operations
	=		equality
	<		less than
	>		greater than

bitwise / logical operations
	!		bitwise not
	&		bitwise and
	|		bitwise or

registers
	w x y z	get from register
	W X Y Z	set to register

array literals
	'		start string
	"		end string
	[		start byte string
	]		end byte string

array manipulation
	L		get array length
	??		resize array
	C		copy byte range
	I		insert byte range
	??		delete byte range

stack-array transfer
	E		big endian
	e		little endian
	o		byte
	p		read from array at offset
	P		write to array at offset
	A		append to array

io
	i		read value	(returns byte, EOF?)
	.		write value
	_		write bytes  (index count --)
