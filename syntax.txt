	' '

data literals
	b		start binary
	B		end binary
	d		start decimal
	D		end decimal
	h		start hexadecimal
	H		end hexadecimal
	`		character
	(		start label
	)		end label

array literals
	'		start string
	"		end string
	[		start byte array
	]		end byte array

state
	{		start state / label
	:		start / end state body
	$		state restart point
	}		end state
	?		start conditional switch state
	;		end switch state

registers
	w x y z	get from register
	W X Y Z	set to register

io
	.		write value
	_		write array
	i		read value	(returns byte, EOF?)

data stack
	t		trim (drop)
	s		swap
	c		copy (dup)
	r		rotate
	R		reverse rotate

array stack
	C		clone nth array
	T		drop nth array
	I		insert top array in nth position
	L		array length / nth array length ?

array stack elements
	#		array concatenate
	S		array split at nth byte

array stack transfer
	E		big endian
	e		little endian
	o		byte
	p		read from top array at offset
	P		write to top array at offset
	a		append to top array
	N		create new array with value

comparison operations
	=		equality
	<		less than
	>		greater than

arithmetic operations
	~		arithmetic negate    <- why is this needed?
	+		addition
	-		subtraction
	*		multiplication
	/		division
	%		modulo

bitwise / logical operations
	!		bitwise not
